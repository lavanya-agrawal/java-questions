Unique digit count

 class UserMainCode{
            public int uniqueDigitCount(int input1){
                    int count=0;
                for(int i=0;i<=9;i++){
        int temp=input1;
        while(temp!=0){
            if(temp%10==i){
                count++;
                break;
            }
            temp/=10;
        }
    }
   
    return count;
            }
        }
