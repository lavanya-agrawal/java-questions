 class UserMainCode{
               public long isPalindrome(string input1){
               input1=input1.toLowerCase();
    int i=0;
    int j= input1.length()-1;
  
    while(i<j){
    char c1 = input1.charAt(i);
    char c2 = input1.charAt(j);
        if(c1!=c2) return 1;
        i++;
        j--;
    }
    return 2;
 	}
}
