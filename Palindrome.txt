Palindrome
class UserMainCode{
            public isPalinNum(int input1){
               
  int rev=0, temp=input1;
    input1 = Math.abs(input1);
    while(input1>0){
        rev = rev*10+input1%10;
        input1/=10;
    }
    if(temp==rev){
        return 2;
    }
    else{
        return 1;
    }
            }
        }
